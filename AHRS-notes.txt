        --privileged ^
        -p 8080:8080 -p 50000:50000 ^
        -v //var/run/docker.sock:/var/run/docker.sock ^
        -v jenkins_home:/var/jenkins_home ^
        jenkins-dind-AHRS

        docker ps
        docker logs jenkins-dind-AHRS

    In Jenkins > copy user admin password paste it in localhost:8080 >
    select install suggested plugings > create first admin user > 
    start using Jenkins

    run:
        docker exec -u root -it jenkins-dind-AHRS bash
        apt update -y
        apt install -y python3
        python3 --version

        python --version

14.2. Github Integration
    In Github > Settings > Developer Settings > Personal access tokens >
    Tokens(classic) > Generate new token (classic) > note: AHRS-token >
    permissions: repo, admin:repo-hook > Generate token > copy token

    In Jenkins > Manage Jenkins > Credentials > global > Add Credentials >
    kind: Username with password > Scope: Global(Jenkins, nodes, items, all child items, etc)>
    username: github user name > password: access token > ID: AHRS:token >
    Description: AHRS-token > create

    In Jenkins > New Item > name: AHRS-MLOPS > Pipeline > 
    Definition: Pipeline script from SCM> SCM:Git > Reposigoty URL: github repo URL >
    credentials: select your credential > branch specifier: */main >
    Apply > Save

    In Jenkins > AHRS-MLOPS > Pipeline Syntax > sample setup: checkout: Check out from version control
    repository URL: github repo URL > select credential > Branch specifier: */main >
    Generate Pipeline Script

    In Jenkinsfile > paste Pipeline script in first stage

    push your local repo to github

    In Jenkins > AHRS-MLOPS > Build Now

14.4. Install Google Cloud and Kubernetes CLI on Jenkins Container
    run:
        docker exec -u root -it jenkins-dind-AHRS bash
        apt-get update
        apt-get install -y curl apt-transport-https ca-certificates gnupg
        curl https://packages.cloud.google.com/apt/doc/apt-key.gpg | apt-key add -
        echo "deb https://packages.cloud.google.com/apt cloud-sdk main" |tee -a /etc/apt/sources.list.d/google-cloud-sdk.list
        apt-get update && apt-get install -y google-cloud-sdk
        apt-get update && apt-get install -y kubectl
        apt-get install -y google-cloud-sdk-gke-gcloud-auth-plugin
        kubectl version --client
        gcloud --version
        exit

        docker restart jenkins-dind-AHRS
